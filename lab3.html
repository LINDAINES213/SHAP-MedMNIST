# ========= 4) Define Simple CNN Model (already provided) =========
import torch.nn as nn
import torch.nn.functional as F

class SimpleCNN(nn.Module):
    def __init__(self, n_channels, n_classes):
        super(SimpleCNN, self).__init__()
        self.conv1 = nn.Conv2d(n_channels, 16, kernel_size=3, padding=1)
        self.pool = nn.MaxPool2d(2, 2)
        self.fc1 = nn.Linear(16*14*14, n_classes)

    def forward(self, x):
        x = self.pool(F.relu(self.conv1(x)))
        x = x.reshape(x.size(0), -1)
        x = self.fc1(x)
        return x

model = SimpleCNN(info['n_channels'], n_classes)
device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
model.to(device)

print("Model class defined and ready.")

def model_forward(x: np.ndarray) -> np.ndarray:
    model.eval()
    with torch.no_grad():
        if isinstance(x, list):
            x = np.stack(x, axis=0)
        x = x.astype(np.float32)

        # Si viene en [0,255], escalar a [0,1]
        if x.max() > 1.0:
            x = x / 255.0

        # Asegurar dimensin batch
        if x.ndim == 3:
            x = x[None, ...]  # (1,H,W,C)

        # Alinear canales esperados vs. provistos
        if x.shape[-1] != info['n_channels']:
            if x.shape[-1] == 1 and info['n_channels'] == 3:
                x = np.repeat(x, 3, axis=-1)  # gris -> RGB
            elif x.shape[-1] == 3 and info['n_channels'] == 1:
                x = x[..., :1]                # RGB -> gris
            else:
                raise ValueError(f"Se esperaban {info['n_channels']} canales, lleg {x.shape[-1]}.")

        # HWC -> CHW y normalizacin (mean=0.5, std=0.5)
        x_t = torch.from_numpy(x).permute(0, 3, 1, 2)
        x_t = (x_t - 0.5) / 0.5
        x_t = x_t.to(device)

        probs = F.softmax(logits, dim=1).cpu().numpy()  # <-- lnea correcta

# ===== Variante B: panel multiclase por imagen + accuracy =====
num_imgs = 5
idxs = np.random.choice(len(test_dataset), size=num_imgs, replace=False)

correct = 0

for k, idx in enumerate(idxs, start=1):
    img_t, true_idx = test_dataset[idx]
    img_np = img_t.permute(1, 2, 0).cpu().numpy().astype(np.float32)
    img_np = np.clip((img_np * 0.5) + 0.5, 0, 1)

    # Prediccin
    probs = model_forward(img_np[None, ...])[0]
    pred_idx = int(probs.argmax())
    pred_prob = float(probs[pred_idx])
    is_correct = (pred_idx == int(true_idx))
    correct += int(is_correct)

    # SHAP
    masker = shap.maskers.Image("blur(28,28)", img_np.shape)
    explainer = shap.Explainer(model_forward, masker, output_names=[id2label[i] for i in range(n_classes)])
    explanation = explainer(img_np[None, ...])

    vals = explanation.values
    # Construimos la lista por clase si es 5D; si es 4D, no hay desagregacin por clase
    if vals.ndim == 5:
        shap_values_list = [vals[..., j] for j in range(n_classes)]  # lista de (1,H,W,C) por clase
        shap.image_plot(shap_values_list, img_np[None, ...], show=False)
    elif vals.ndim == 4:
        shap.image_plot(vals, img_np[None, ...], show=False)
    else:
        raise ValueError(f"Dimensin inesperada en explanation.values: {vals.shape}")

    # Aadimos un ttulo global informativo a la figura recin generada por SHAP
    fig = plt.gcf()
    check = "" if is_correct else ""
    fig.suptitle(
        f"Imagen {k}/{num_imgs}  Real: {id2label[int(true_idx)]} | "
        f"Pred: {id2label[pred_idx]} ({pred_prob:.0%}) {check}",
        fontsize=12
    )
    plt.show()

# Accuracy global al final
acc = correct / num_imgs
print(f"Aciertos: {correct}/{num_imgs}  Accuracy: {acc:.2%}")
